class UnionFind:
    def __init__(self, n):
        self.parents = {}
        self.rank = {}

        for i in range(1, n + 1):
            self.parents[i] = i
            self.rank[i] = 0

    def find(self, nd):
        p = self.parents[nd]

        while p != self.parents[p]:
            self.parents[p] = self.parents[self.parents[p]]
            p = self.parents[p]

        return p
